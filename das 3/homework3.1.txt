package com.company;

     //Write a program to enter the numbers till the user wants and at the end it should display the count of positive,
     //negative and zeros entered.

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int countOfPositiveNumbers = 0;
        int countOfNegativeNumbers = 0;
        int countOfZero = 0;
        boolean selectedNumber;
        do{
            System.out.print("Input number  ");
            int inputNumber = scanner.nextInt();
            if (inputNumber > 0)
            {
                countOfPositiveNumbers++;
            }
            else if (inputNumber < 0)
            {
                countOfNegativeNumbers++;
            }
            else
            {
                countOfZero++;
            }
            System.out.print("Do you want to continue (true or false)?   ");
            selectedNumber = scanner.nextBoolean();
        }
        while(selectedNumber);

        System.out.println("You inputted " + countOfPositiveNumbers + " positive number.");
        System.out.println("You inputted " + countOfNegativeNumbers + " negative number.");
        System.out.println("You inputted " + countOfZero + " zero.");
    }
}
