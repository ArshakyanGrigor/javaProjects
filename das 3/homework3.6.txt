package com.company;

     // Write a program to enter the numbers till the user wants and at the end the program should display the largest
     // and smallest numbers entered.
     // Every time user enters number ask â€œDo you want to continue true/n? â€.

import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int inputNumber;
        int largestNumber =  Integer.MIN_VALUE;
        int smallestNumber = Integer.MAX_VALUE;
        boolean selectedNumber = true;
        do
        {
            System.out.print("Input number ");
            inputNumber = scanner.nextInt();
            if (inputNumber > largestNumber)
            {
                largestNumber = inputNumber;
            }
            if (inputNumber < smallestNumber)
            {
                smallestNumber = inputNumber;
            }
            System.out.print("Do you want to continue (true or false)  ");
            selectedNumber = scanner.nextBoolean();
        }
        while (selectedNumber);

        System.out.println("The largest number is " + largestNumber);
        System.out.println("The smallest number is " + smallestNumber);
    }
}
